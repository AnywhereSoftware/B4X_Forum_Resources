Sub Handle (req As ServletRequest, resp As ServletResponse)
	Request = req
	Response = resp
	Method = Request.Method
	Log($"${Request.Method}: ${Request.RequestURI}"$)
	Dim path As String = req.RequestURI
	If path = "/$endpoints$" Then
		RenderPage
	Else If path = "/hx/$endpoints$/table" Then
		HandleTable
	Else If path = "/hx/$endpoints$/add" Then
		HandleAddModal
	Else If path.StartsWith("/hx/$endpoints$/edit/") Then
		HandleEditModal
	Else If path.StartsWith("/hx/$endpoints$/delete/") Then
		HandleDeleteModal
	Else
		Handle$Endpoints$
	End If
End Sub

' Handle CRUD operations
Private Sub Handle$Endpoints$
	Select Method
		Case "POST"
			' Create
			Dim name As String = Request.GetParameter("name")
			If name = "" Or name.Trim.Length < 2 Then
				ShowAlert("$Endpoint$ name must be at least 2 characters long.", "warning")
				Return
			End If
			Try
				DB.SQL = DB.Open
				DB.Table = "$TableName$"
				DB.Conditions = Array("$endpoint$_name = ?")
				DB.Parameters = Array(name)
				DB.Query
				If DB.Found Then
					ShowAlert("$Endpoint$ already exists!", "warning")
					DB.Close
					Return
				End If
			Catch
				Log(LastException)
				ShowAlert($"Database error: ${LastException.Message}"$, "danger")
			End Try

			' Insert new row
			Try
				DB.Reset
				DB.Columns = Array("$endpoint$_name", "created_date")
				DB.Parameters = Array(name, Main.CurrentDateTime)
				DB.Save
				ShowToast("$Endpoint$", "created", "$Endpoint$ created successfully!", "success")
			Catch
				ShowAlert($"Database error: ${LastException.Message}"$, "danger")
			End Try
			DB.Close
		Case "PUT"
			' Update
			Dim id As Int = Request.GetParameter("id")
			Dim name As String = Request.GetParameter("name")
			DB.SQL = DB.Open
			DB.Table = "$TableName$"
			
			DB.Find(id)
			If DB.Found = False Then
				ShowAlert("$Endpoint$ not found!", "warning")
				DB.Close
				Return
			End If

			DB.Reset
			DB.Conditions = Array("$endpoint$_name = ?", "id <> ?")
			DB.Parameters = Array(name, id)
			DB.Query
			If DB.Found Then
				ShowAlert("$Endpoint$ already exists!", "warning")
				DB.Close
				Return
			End If
			
			' Update row
			Try
				DB.Reset
				DB.Columns = Array("$endpoint$_name", "modified_date")
				DB.Parameters = Array(name, Main.CurrentDateTime)
				DB.Id = id
				DB.Save
				ShowToast("$Endpoint$", "updated", "$Endpoint$ updated successfully!", "info")
			Catch
				ShowAlert($"Database error: ${LastException.Message}"$, "danger")
			End Try
			DB.Close
		Case "DELETE"
			' Delete
			Dim id As Int = Request.GetParameter("id")
			DB.SQL = DB.Open
			DB.Table = "$TableName$"
			
			DB.Find(id)
			If DB.Found = False Then
				ShowAlert("$Endpoint$ not found!", "warning")
				DB.Close
				Return
			End If
			
			DB.Table = "tbl_products"
			DB.WhereParam("$endpoint$_id = ?", id)
			DB.Query
			If DB.Found Then
				ShowAlert("Cannot delete $endpoint$ with associated products!", "warning")
				DB.Close
				Return
			End If

			' Delete row
			Try
				DB.Table = "$TableName$"
				DB.Id = id
				DB.Delete
				ShowToast("$Endpoint$", "deleted", "$Endpoint$ deleted successfully!", "danger")
			Catch
				ShowAlert($"Database error: ${LastException.Message}"$, "danger")
			End Try
			DB.Close
	End Select
End Sub

Private Sub Create$Endpoints$Table As MiniHtml
	If App.ctx.ContainsKey("/hx/$endpoints$/table") = False Then
		Dim table1 As MiniHtml = Table.cls("table table-bordered table-hover rounded small")
		Dim thead1 As MiniHtml = Thead.up(table1).cls("table-light")
		Th.up(thead1).text("#").sty("text-align: right; width: 50px")
		Th.up(thead1).text("Name")
		Th.up(thead1).text("Actions").sty("text-align: center; width: 120px")
		Tbody.up(table1)
		App.ctx.Put("/hx/$endpoints$/table", table1)
	End If
	
	DB.SQL = DB.Open
	DB.Table = "$TableName$"
	DB.Columns = Array("id", "$endpoint$_name")
	DB.OrderBy = CreateMap("id": "")
	DB.Query

	Dim table1 As MiniHtml = App.ctx.Get("/hx/$endpoints$/table")
	Dim tbody1 As MiniHtml = table1.Child(1)
	tbody1.Children.Clear ' remove all children
	For Each row As Map In DB.Results
		Dim tr1 As MiniHtml = Create$Endpoints$Row		
		tr1.Child(0).text2(row.Get("id"))
		tr1.Child(1).text2(row.Get("$endpoint$_name"))
		tr1.Child(2).Child(0).attr("hx-get", "/hx/$endpoints$/edit/" & row.Get("id"))
		tr1.Child(2).Child(1).attr("hx-get", "/hx/$endpoints$/delete/" & row.Get("id"))
		tr1.up(tbody1)
	Next
	DB.Close
	Return table1
End Sub

Private Sub ShowToast (entity As String, action As String, message As String, status As String)

	"action": action, _

	"status": status))
