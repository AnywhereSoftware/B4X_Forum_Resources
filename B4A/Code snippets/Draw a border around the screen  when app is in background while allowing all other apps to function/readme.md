### Draw a border around the screen  when app is in background while allowing all other apps to function normally - courtesy of Claude Code by JackKirk
### 06/10/2026
[B4X Forum - B4A - Code snippets](https://www.b4x.com/android/forum/threads/171197/)

This code allows you to draw a border (in this case green) around the screen when app is in background while allowing all other apps to function normally.  
  
My use case was the desire to flag the state of a background running app while it is not visible.  
  
Code primarily generated by Claude Code with extensive manual tidy up.  

```B4X
Private Sub Process_Globals  
  
    Private Obj_strip As JavaObject                  
    Private Obj_strip_wm As JavaObject  
              
    Private Gen_strip_visible As Boolean  
   
    Private GRAVITY_TOP    As Int = 48  
    Private GRAVITY_BOTTOM As Int = 80  
    Private GRAVITY_LEFT   As Int = 3  
    Private GRAVITY_RIGHT  As Int = 5  
    Private MATCH_PARENT As Int = -1  
  
    'TYPE_APPLICATION_OVERLAY requires SYSTEM_ALERT_WINDOW permission - draws above  
    'all other windows including other apps  
    Private TYPE_APPLICATION_OVERLAY As Int = 2038  
   
    'Android's internal integer code for pixel format tells WindowManager what pixel  
    'format to use for overlay window, can be:  
    '0 = UNKNOWN  
    '1 = RGBA_8888  
    '2 = RGBX_8888 (no alpha channel)  
    '3 = RGB_888  
    '4 = RGB_565  
    '  
    'RGBA_8888 means each pixel has 4 channels, each 8 bits:  
    '- **R**ed — 8 bits (0-255)  
    '- **G**reen — 8 bits (0-255)  
    '- **B**lue — 8 bits (0-255)  
    '- **A**lpha — 8 bits (0-255, where 0=fully transparent, 255=fully opaque)  
    'Total = 32 bits per pixel  
    '  
    'Alpha channel is what makes transparent FrameLayout actually transparent — without  
    'it, "transparent" background would render as black  
    '  
    'RGBA_8888 tells Android to honour alpha values when compositing overlay onto whatever  
    'is behind it  
    Private RGBA_8888 As Int = 1  
  
    'NOT_FOCUSABLE - strip does not steal keyboard/IME focus  
    Private NOT_FOCUSABLE As Int = 8  
   
    'NOT_TOUCHABLE  - all touches pass through strip to app below  
    Private NOT_TOUCHABLE As Int = 16  
  
End Sub  
  
'************************************************************************************  
'  
'This procedure shows a thin green strip around screen via WindowManager overlay  
'visible over all apps  
'  
'Input parameters are:  
'  
'       None  
'  
'Returns:  
'  
'       -  
'  
'Notes on this procedure:  
'  
'       o None  
'  
'************************************************************************************  
Private Sub Strip_Show  
   
    Try  
                                                                                     
        Private wrk_jo_context As JavaObject                                                    
        wrk_jo_context.InitializeContext                                                    
     
        'Density-based thickness ensures strip is physically consistent across screen  
        'densities, Max(4, …) ensures at least 4 physical pixels on very low  
        'density screens  
        Private wrk_thick As Int = Max(4, DipToCurrent(4))                                  
     
        'Full-screen transparent FrameLayout - children positioned by gravity        
        Obj_strip.InitializeNewInstance("android.widget.FrameLayout", Array(wrk_jo_context))  
        Obj_strip.RunMethod("setBackgroundColor", Array(Colors.Transparent))                  
  
        'Top strip                                                                    
        Private wrk_top As JavaObject                                                    
        wrk_top.InitializeNewInstance("android.view.View", Array(wrk_jo_context))          
        wrk_top.RunMethod("setBackgroundColor", Array(Colors.Green))                  
        Private wrk_top_lp As JavaObject  
        wrk_top_lp.InitializeNewInstance("android.widget.FrameLayout$LayoutParams", _  
            Array(MATCH_PARENT, wrk_thick, GRAVITY_TOP))                                          
        wrk_top.RunMethod("setLayoutParams", Array(wrk_top_lp))                      
        Obj_strip.RunMethod("addView", Array(wrk_top))                              
  
        'Bottom strip                                                                  
        Private wrk_bot As JavaObject                                                    
        wrk_bot.InitializeNewInstance("android.view.View", Array(wrk_jo_context))          
        wrk_bot.RunMethod("setBackgroundColor", Array(Colors.Green))                  
        Private wrk_bot_lp As JavaObject                                                  
        wrk_bot_lp.InitializeNewInstance("android.widget.FrameLayout$LayoutParams", _  
            Array(MATCH_PARENT, wrk_thick, GRAVITY_BOTTOM))                                                  
        wrk_bot.RunMethod("setLayoutParams", Array(wrk_bot_lp))                      
        Obj_strip.RunMethod("addView", Array(wrk_bot))                              
  
        'Left strip                                                                    
        Private wrk_lft As JavaObject                                                    
        wrk_lft.InitializeNewInstance("android.view.View", Array(wrk_jo_context))          
        wrk_lft.RunMethod("setBackgroundColor", Array(Colors.Green))                  
        Private wrk_lft_lp As JavaObject                                                  
        wrk_lft_lp.InitializeNewInstance("android.widget.FrameLayout$LayoutParams", _  
            Array(wrk_thick, MATCH_PARENT, GRAVITY_LEFT))                                                
        wrk_lft.RunMethod("setLayoutParams", Array(wrk_lft_lp))                      
        Obj_strip.RunMethod("addView", Array(wrk_lft))                              
  
        'Right strip                                                                  
        Private wrk_rgt As JavaObject                                                    
        wrk_rgt.InitializeNewInstance("android.view.View", Array(wrk_jo_context))          
        wrk_rgt.RunMethod("setBackgroundColor", Array(Colors.Green))                  
        Private wrk_rgt_lp As JavaObject                                                  
        wrk_rgt_lp.InitializeNewInstance("android.widget.FrameLayout$LayoutParams", _  
            Array(wrk_thick, MATCH_PARENT, GRAVITY_RIGHT))                                                
        wrk_rgt.RunMethod("setLayoutParams", Array(wrk_rgt_lp))                      
        Obj_strip.RunMethod("addView", Array(wrk_rgt))                              
  
        'Full-screen window - TYPE_APPLICATION_OVERLAY, NOT_FOCUSABLE, NOT_TOUCHABLE,  
        'RGBA_8888 - all touches pass through  
        Private wrk_lp As JavaObject                                                    
        wrk_lp.InitializeNewInstance("android.view.WindowManager$LayoutParams", _  
            Array(MATCH_PARENT, MATCH_PARENT, TYPE_APPLICATION_OVERLAY, NOT_FOCUSABLE + NOT_TOUCHABLE, RGBA_8888))  
        Obj_strip_wm.InitializeContext                                              
        Obj_strip_wm = Obj_strip_wm.RunMethod("getSystemService", Array("window"))  
        Obj_strip_wm.RunMethod("addView", Array(Obj_strip, wrk_lp))                
  
        'Flag green strip visible  
        Gen_strip_visible = True  
  
    Catch                                    
        Log("Strip_Show: " & LastException.Message)            
    End Try                                    
  
End Sub                                
  
'************************************************************************************  
'  
'This procedure removes the green strip overlay added by Strip_Show  
'  
'Input parameters are:  
'  
'       None  
'  
'Returns:  
'  
'       -  
'  
'Notes on this procedure:  
'  
'       o None  
'  
'************************************************************************************  
Private Sub Strip_Hide  
  
    Try  
        Obj_strip_wm.RunMethod("removeView", Array(Obj_strip))  
    Catch  
        Log("Strip_Hide: " & LastException.Message)  
    End Try  
  
    'Flag green strip hidden  
    Gen_strip_visible = False  
  
End Sub
```

  
and you need this in the Manifest:  
  
AddPermission(android.permission.SYSTEM\_ALERT\_WINDOW)
